    jc.name AS category_name,
    COUNT(DISTINCT fc.id) AS comment_count,
    COUNT(DISTINCT CASE WHEN fc.level = 1 THEN fc.id END) AS top_level,
    COUNT(DISTINCT CASE WHEN fc.level = 2 THEN fc.id END) AS replies
FROM job_category_simple jc
LEFT JOIN forum_comments fc ON jc.next_category_id = fc.category_id
GROUP BY jc.id, jc.name
